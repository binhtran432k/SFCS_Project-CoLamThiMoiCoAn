CREATE TABLE "Users" (
	"UserID"	INTEGER PRIMARY KEY AUTOINCREMENT,
	"Email"	TEXT NOT NULL UNIQUE,
	"LoginName"	TEXT UNIQUE,
	"Hash"	TEXT NOT NULL,
	"UserName"	TEXT NOT NULL,
	"UserType"	INTEGER NOT NULL,
	"CreatedDate"	INTEGER NOT NULL,
	"Valid"	INTEGER NOT NULL
);

CREATE TABLE "Stalls" (
	"OwnerID"	INTEGER NOT NULL UNIQUE,
	"StallName"	TEXT NOT NULL UNIQUE,
	"StallDescription"	TEXT,
	FOREIGN KEY("OwnerID") REFERENCES "Users"("UserID")
);

CREATE TABLE "Cooks" (
	"CookID"	INTEGER NOT NULL UNIQUE,
	"OwnerID"	INTEGER NOT NULL,
	FOREIGN KEY("OwnerID") REFERENCES "Users"("UserID"),
	FOREIGN KEY("CookID") REFERENCES "Users"("UserID")
);

CREATE TABLE "Foods" (
	"FoodID"	INTEGER PRIMARY KEY AUTOINCREMENT,
	"OwnerID"	INTEGER NOT NULL,
	"FoodName"	TEXT NOT NULL,
	"FoodPrice"	INTEGER,
	"OriginalPrice"  INTEGER,
	"FoodState"	INTEGER NOT NULL,
	"FoodDescription"	TEXT,
	"LastUpdatedDate"	INTEGER NOT NULL,
	FOREIGN KEY("OwnerID") REFERENCES "Users"("UserID")
);

CREATE TABLE "Orders" (
	"OrderID"	INTEGER PRIMARY KEY AUTOINCREMENT,
	"OwnerID"	INTEGER NOT NULL,
	"CustomerID"	INTEGER NOT NULL,
	"CookID"   INTEGER,
	"OrderState"	INTEGER NOT NULL,
	"CreatedDate"	INTEGER NOT NULL,
	FOREIGN KEY("CustomerID") REFERENCES "Users"("UserID"),
    FOREIGN KEY("CookID") REFERENCES "Users"("UserID"),
	FOREIGN KEY("OwnerID") REFERENCES "Users"("UserID")
);

CREATE TABLE "Carts" (
	"CartID"	INTEGER PRIMARY KEY AUTOINCREMENT,
	"OrderID"	INTEGER,
	"FoodID"	INTEGER NOT NULL,
	"CartState"	INTEGER NOT NULL,
	FOREIGN KEY("FoodID") REFERENCES "Foods"("FoodID"),
	FOREIGN KEY("OrderID") REFERENCES "Orders"("OrderID")
);
